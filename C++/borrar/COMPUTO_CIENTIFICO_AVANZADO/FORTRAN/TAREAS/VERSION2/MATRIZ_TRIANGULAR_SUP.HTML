ESPACIO open(unit=1,file=arch_matriz, status="old", action="read",err=99) !Lee la dimensión de la matriz read(1,*,iostat=ios)n !si no se encuentra el archivo o este esta vacío se produce un error if (ios/=0) then close(1) stop "ERROR: NO HUBO LECTURA (!Verifique que el archivo a leer exista!)" end if if (n<=1) then close(1) stop "ERROR: DIMENSI¢N MENOR O IGUAL A 1" end if !Imprime el valor de n leido print '(16x,(a),\)', 'DIMENSI¢N:' print *, n !************************* MATRIZ TRIANGULAR INFERIOR***********************! !reserva espacio para a allocate(a(n)) do i=1,n allocate(a(i)%r(1:i)) end do !Lectura de la matriz triangular inferior do i=1,n read(1,*,iostat=ios)a(i)%r(1:i) !lee cada una de las líneas de archivo como !una fila de la matriz a if (ios/=0) stop "ERROR: error de lectura"!error de lectura !ERROR: DIAGONALES = 0 if (a(i)%r(i)==0) stop "ERROR: Los elementos de la diagonal deben ser distintos de cero" end do !Imprimo para verificar si todo ok!!
